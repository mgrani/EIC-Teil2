  \begin{bsslide}[\textbufferB]
    \colortext{Farbhistogramm}
    \bspar1
    \begin{definition}[Farbhistogramm]
      Ein \textbf{Farbhistogramm $h_c(i)$} stellt die Anzahl der Pixel für einen spezifischen Farbwert $i$ in jedem Kanal $c$ eines Farbraumes dar
    \end{definition}
    \begin{itemize}
      \item Bei Grauwertbildern spricht man auch vom Intensitätshistogramm
      \item Erkennen von Grundeigenschaften von Bildern (e.g. Kontrastarme Bilder sind meist im Luminanz/Grauwertbereich limitiert)
      \item Grundlagen der meisten Punktoperationen (e.g. Farbanpassung, Aufhellung)
      \item Anwendbar auf jeden Farbkanal od. Grauwertkanal
      \item Kumulativen Histogramms $H_c(i)=\sum_i h_c(i)$ = Summation der H\"aufigkeiten der Grauwerte $i$
    \end{itemize}
    \bspar1
    
  \end{bsslide}

  \begin{bsslide}[\textbufferB]
    \colortext{Histogramm Einebnung/Histogrammausgleich/Histogrammäqualisation}
    \bspar1\small
    \textbf{Ziel}: möglichst gleichmäßige Verteilung der Intensitätsstufen
    \begin{itemize}
      \item Betrachtung des kumulativen Grauwerthistogramms $H(i)$ (entspricht der Anzahl der Pixel mit Grauwert < i)
      \item Berechnung einer Lookup-Tabelle $G'(i)$ für Grau-/Farbwerte $i$:
      \[
      G'(i) = round(\frac{G_{max}}{T}\cdot H(i))
      \]
      \item $G_{max}$ maximal möglicher Grauwert, $T$ Anzahl der Pixel
      \item $G'(i)$ entspricht dem neuen Grauwert für vorigen Grauwert $i$
      \item[$\rightarrow$] Entspricht einer linearen Interpolation des kumulativen Histogramms
    \end{itemize}
    \bsfigurecaption[0.4]{bild-op-schema-histogramm-einebnung}{Schematische Darstellung Histogrammeinebnung (Bildquelle Wikipedia)}
  \end{bsslide}

  \begin{bsslide}[\textbufferB]
    \colortext{Filter - Allgemein} 
    \bspar1
    Bei jeder Nachbarschaft kommt es zum Randproblem
    \bsfigurecaption[0.7]{bild-op-nachbarschaft-rand.png}{}
    Lösungen
    \begin{itemize}
      \item Randpixel werden nicht betrachtet. D.h. aber das das Bild bei jeder Operation schrumpft
      \begin{itemize}
      \item Nachbarschaft wird verkleinert (d.h. Filter wird ungenauer)
      \item Extrapolation - Extrapolationsfehler
      \item Periodische Fortsetzung des Bildes - nicht bei allen Bildern eine gültige Annahme
      \end{itemize}
    \end{itemize}

  \begin{bsslide}[\textbufferB]
    \colortext{Lineare Filter}
    \bspar1
    \textbf{Linearer Filter:} Jeder Pixelwert wird durch einen linear Kombination der Pixelwerte in seiner Nachbarschaft ermittelt
    \begin{itemize}
  \small
  \item Die Nachbarschaft wird auch als \textbf{Kernelregion oder Kernel K} bezeichnet
  \item Die Multiplikation des Originalbilds mit dem Kernel ist eine \textbf{Faltung} 
\end{itemize}
    \bsfigurecaption[0.4]{bild-op-kernel}{}
    \[
    B^{Neu}(x,y)=K*B(x,y)=\begin{pmatrix}
    0 & a & 0\\
    b & c & d \\
    0 & e & 0
    \end{pmatrix}
    =\sum_{i=-M}^{M}\sum_{j=-N}^{N} K(i,j)*B(x-i,y-j)
    \]
  \end{bsslide}

  \item Effiziente Implementierung für große Filter möglich

\begin{bsslide}[\textbufferB]
  \colortext{Nicht-lineare Filter}
  \bspar1
  Lineare Operatoren besitzen in manchen Fällen Nachteile: So unterdrückt ein linearer Glättungsoperator zwar das Rauschen, aber er verursacht gleichzeitig Unschärfe bei den Kanten. Man benutzt daher auch nichtlineare Filter, von denen die Rangordnungsoperatoren einen wichtigen Typ darstellen.\bspar2
  \textbf{Rangordnungsoperatoren}:  Rangordnungsoperatoren ordnen zunächst die Nachbarpixel eines Bildpunktes in einer aufsteigenden Folge und ermitteln einen neuen Grauwert aufgrund einer Position in der Reihe
  \bspar2
  Beispiele nicht-lineare Filter
  \begin{itemize}
    \item Medianfilter 
    \item Morphologische Operationen (Erosion, Dilatation)
     
  \end{itemize}
\end{bsslide}

\begin{bsslide}[\textbufferB]
  \colortext{Erosion} 
  \bspar1
  Notation: $A\ominus B$ (Bild minus Maske, Aufbrechen geschlossener Bereiche)
  \begin{itemize}
    \item Ausgangspunkt: Binärbild (i.e. Pixel ist 0 oder 1)
    \item Definition einer speziellen Strukturmaske (z.B. Kreis)
    
    \item Für jedes Pixel im Bild wird nun geprüft, ob das Strukturelement vollständig in den Pixel passt (d.h. alle Pixel der Strukturmaske um den Pixel 1 sind)
    \item Gilt dies für Pixel nicht, wird der Pixel aus dem Bild entfernt
  \end{itemize} 
  \bspar2
  \bsfigurecaption{bild-op-erosion}{\tiny Bildquelle Wikipedia}
  
\end{bsslide}

\begin{bsslide}[\textbufferB]
  \colortext{Dilatation}
  \bspar1
  Notation: $A\oplus B$ (Bild plus Maske, Schließen offener Bereiche)
  \begin{itemize} 
    \item Ausgangspunkt: Binärbild (i.e. Pixel ist 0 oder 1)
    \item Definition einer speziellen Strukturmaske (z.B. Kreis)
    \item Für jedes Pixel mit Wert 1 im Bild, werden alle Pixel, welche von der Strukturmaske besetzt sind, auf 1 gesetzt
  \end{itemize}
  \bspar2
  \bsfigurecaption[0.8]{bild-op-dilatation}{\tiny Bildquelle Wikipedia}

  \begin{bsslide}[\textbufferB]
    \colortext{Geometrische Transformationen}
    \bspar1
    Punkt- und Filteroperationen belassen Pixel noch auf ihren Platz\bspar2
    \textbf{Geometrische Transformationen } verschieden die Pixelposition
    \begin{itemize}
      \item Verschiebung (Translation)
      \item Streckung/Stauchung (Verkleinerung/Vergrößerung)
      \item Drehung (Rotation)
      \item Scherung (Parallelverschiebungen)
    \end{itemize}
    \textbf{Affine Transformation:}
    \begin{itemize}
      \item Beschränkte Objekte bleiben beschränkt.
      \item Parallele Objekte bleiben parallel
      \item Längen-, Flächen- und Volumenverhältnisse bleiben erhalten.
    \end{itemize}
  \end{bsslide}

    \colortext{Geometrische Transformationen - Stauchung auf Rasterbildern}
    \bspar1
    \textbf{Problem bei Stauchung:} Nicht jedes Ausgangspixel kann genau einem Zielpixel zugeordnet werden
    \bspar1
    Lösung z.B. über Mittelwert-Bildung, wenn mehrere Pixel auf ein Pixel abgebildet werden sollen

    \bspar1
    Verhindern des Morie-Effeks durch Weichzeichner vor Skalierung (=Tiefpassfilter)\bspar1
  \end{bsslide}
